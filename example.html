Quick Start

Get OpenScroll running in your project with just a few lines of code:

HTML
// Include the library
<script src="src/OpenScroll.js"></script>
<script src="src/openAnimations/SmoothScroller.js"></script>
JavaScript
// Basic usage
document.addEventListener('DOMContentLoaded', () => {
    const openScroll = new OpenScroll();

    // Start smooth scrolling
    openScroll.start(SmoothScroller, {
        element: '#scrollSmooth',
        smoothness: 0.95,
        speed: 1.2
    });
});
JavaScript
// Load by string from global scope
document.addEventListener('DOMContentLoaded', (event) => {
    const openScroll = new OpenScroll();

    const smoothScroller = openScroll.start('SmoothScroller');

    // Check what's running
    console.log('Running openAnimations:', openScroll.getAllInstances());
});
